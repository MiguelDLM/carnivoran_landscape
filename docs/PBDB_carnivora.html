Taxonomy

Do a request to the PBDB API to retrieve taxonomy information for the filtered taxa.

Code 
## Build taxonomy URL safely: remove accidental spaces in the `show` parameter
show_fields <- "attr,app,common,parent,immparent,classext"
taxonomy_url <- paste0(
  "https://paleobiodb.org/data1.2/taxa/list.csv?name=",
  URLencode(group),
  "&rel=all_children&show=",
  URLencode(show_fields)
)

message("Downloading PBDB taxonomy data from: ", taxonomy_url)
taxonomy_data <- tryCatch(
  read_csv(taxonomy_url, show_col_types = FALSE),
  error = function(e) {
    stop("Error downloading PBDB taxonomy data: ", e$message)
  }
)
Code 
# Filter to keep only species-level taxa
taxonomy_filtered <- taxonomy_data %>%
  filter(accepted_rank == "species") %>%
  distinct(accepted_name, .keep_all = TRUE)
Code 
# Enriquecimiento de `taxonomy_filtered` con niveles taxonómicos obtenidos por taxize
library(purrr)
library(dplyr)
library(tidyr)

# Vector de nombres a consultar (especies filtradas)
taxa_vector <- taxonomy_filtered$accepted_name

# Función auxiliar: intenta varios servicios hasta obtener un resultado
get_classification_safe <- function(taxon) {
  dbs <- c("itis", "ncbi", "gbif")
  for (db in dbs) {
    res <- tryCatch(taxize::classification(taxon, db = db), error = function(e) NULL)
    if (!is.null(res) && length(res) > 0) {
      # taxize puede devolver lista; extraer primer elemento si es necesario
      df <- if (is.list(res) && !is.data.frame(res)) res[[1]] else res
      if (is.data.frame(df) && nrow(df) > 0) return(df)
    }
  }
  return(NULL)
}

# Rangos que queremos como columnas
ranks_want <- c("kingdom", "phylum", "class", "order", "family", "genus", "species")

# Consulta todas las especies (puede tardar; considera cachear)
upstream_list <- purrr::map(taxa_vector, get_classification_safe)
Code 
# Convertir la lista de clasificaciones a un dataframe ancho (una fila por taxón)
tax_df <- purrr::map2_dfr(taxa_vector, upstream_list, ~{
  taxon_name <- .x
  cl <- .y
  if (is.null(cl) || nrow(cl) == 0) {
    # fila vacía con solo el nombre
    tibble(accepted_name = taxon_name)
  } else {
  }
})

# Hacer left join para enriquecer taxonomy_filtered
taxonomy_enriched <- taxonomy_filtered %>%
  left_join(tax_df, by = "accepted_name")

if (!"Species2" %in% names(mandibles_data) && all(c("Genus","Species") %in% names(mandibles_data))) {

}

# Crear columna availability comprobando membership fuera de mutate evitando NSE
mandibles_dataset <- taxonomy_enriched %>%
  mutate(availability = ifelse(accepted_name %in% mandibles_data$Species2, "Chatar", ""))

Interactive Sunburst Chart

Code 
# Interactive sunburst chart (3-level): Root > Family > Availability > Extant Status
library(plotly)
library(dplyr)

# Prepare data for sunburst: ensure we have the columns and clean values
mandibles_dataset <- mandibles_dataset %>%
  mutate(
    family = ifelse(is.na(family.y) | family.y == "", "Unknown", as.character(family.y)),
    availability = ifelse(is.na(availability) | availability == "", "No Data", 
                         ifelse(availability == "Chatar", "Chatar", "No Data")),
    is_extant = ifelse(is.na(is_extant) | is_extant == "", "Unknown", as.character(is_extant))
  )

# Build hierarchical data structure for sunburst
# Level 1: Root (center)
# Level 2: Families 
# Level 3: Availability within families
# Level 4: Extant status within availability

# Count species at each level
species_counts <- mandibles_dataset %>%
  group_by(family, availability, is_extant) %>%
  summarise(n_species = n_distinct(accepted_name), .groups = "drop") %>%
  filter(n_species > 0)

# Create sunburst data structure
sunburst_data <- tibble(
  ids = character(),
  labels = character(), 
  parents = character(),
  values = numeric(),
  colors = character()
)

# Root node
sunburst_data <- sunburst_data %>%
  add_row(ids = "Carnivora", labels = "Carnivora", parents = "", 
          values = sum(species_counts$n_species), colors = "#CCCCCC")

# Family level nodes
family_data <- species_counts %>%
  group_by(family) %>%
  summarise(total_species = sum(n_species), .groups = "drop") %>%
  arrange(desc(total_species))

for(fam in family_data$family) {
  total_sp <- family_data$total_species[family_data$family == fam]
  sunburst_data <- sunburst_data %>%
    add_row(ids = fam, labels = paste0(fam, "<br>(", total_sp, " spp)"), 
            parents = "Carnivora", values = total_sp, colors = "#888888")
}

# Availability level nodes  
avail_data <- species_counts %>%
  group_by(family, availability) %>%
  summarise(total_species = sum(n_species), .groups = "drop")

for(i in 1:nrow(avail_data)) {
  fam <- avail_data$family[i]
  avail <- avail_data$availability[i]
  total_sp <- avail_data$total_species[i]
  node_id <- paste(fam, avail, sep = " - ")
  
  color <- ifelse(avail == "Chatar", "#228B22", "#DC143C")  # Green for Chatar, Red for No Data
  
  sunburst_data <- sunburst_data %>%
    add_row(ids = node_id, labels = paste0(avail, "<br>(", total_sp, " spp)"),
            parents = fam, values = total_sp, colors = color)
}

# Extant status level nodes (leaf nodes)
for(i in 1:nrow(species_counts)) {
  fam <- species_counts$family[i]
  avail <- species_counts$availability[i]
  extant <- species_counts$is_extant[i]
  n_sp <- species_counts$n_species[i]
  
  parent_id <- paste(fam, avail, sep = " - ")
  node_id <- paste(fam, avail, extant, sep = " - ")
  
  # Colors for extant status
  color <- case_when(
    extant == "extant" ~ "#4169E1",    # Blue for extant
    extant == "extinct" ~ "#FFD700",   # Gold for extinct  
    TRUE ~ "#D3D3D3"                   # Light gray for unknown
  )
  
  sunburst_data <- sunburst_data %>%
    add_row(ids = node_id, labels = paste0(extant, "<br>(", n_sp, " spp)"),
            parents = parent_id, values = n_sp, colors = color)
}

# Create interactive sunburst chart
fig <- plot_ly(
  type = "sunburst",
  ids = sunburst_data$ids,
  labels = sunburst_data$labels, 
  parents = sunburst_data$parents,
  values = sunburst_data$values,
  branchvalues = "total",
  hovertemplate = "<b>%{label}</b><br>Species: %{value}<br>Percentage: %{percentParent}<extra></extra>",
  maxdepth = 4,
  insidetextorientation = 'radial'
) %>%
  layout(
    title = list(
      text = "Interactive Carnivoran Species Distribution",
      font = list(size = 16)
    ),
    font = list(size = 12),
    margin = list(t = 50, l = 0, r = 0, b = 0)
  )

# Enriquecer mandibles_dataset con información de Ann (similar a como se hizo con Chatar)
# Crear vector de especies de Ann para búsqueda rápida
ann_species <- unique(ann_data$Species)

# Añadir información de Ann a mandibles_dataset
mandibles_dataset <- mandibles_dataset %>%
  mutate(
    # Marcar si la especie está disponible en la base de datos de Ann
    ann_available = case_when(
      accepted_name %in% ann_species ~ "Ann",
      TRUE ~ ""
    ),
    # Actualizar columna availability: si ya tenía "Chatar" y también está en Ann, crear nota
    note = case_when(
      availability == "Chatar" & ann_available == "Ann" ~ "Also available in Ann",
      TRUE ~ NA_character_
    ),
    # Actualizar availability: priorizar Ann si está disponible, sino mantener valor original
    availability = case_when(
      ann_available == "Ann" ~ "Ann",
      TRUE ~ availability
    )
  )
